def jieba_tokenizer(text):
    words = pseg.cut(text)
    return ' '.join([ 
        word for word, flag in words if flag != 'x'])

import numpy as np
def vectorize(words):
    self_main_list = [0] * len(vectorterms)
    for term in words:
        if term in vectorterms:  # 
            idx = vectorterms.index(term)
            self_main_list[idx] += 1
    return np.array(self_main_list)

# Tokenized
