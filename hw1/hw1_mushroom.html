# calculate prior probablity, likelihood
def cal_prior_prob(label, y_train):
    total_case_num = y_train.shape[0]
    class_classify = np.sum(y_train==label)
    
    return class_classify/float(total_case_num)

def cal_likelihood(x_train, y_train, feature_col, feature_variation, label):
    x_current = x_train[y_train==label]
    deno = np.sum(y_train==label)
    nume = np.sum(x_current[:,feature_col]==feature_variation)
    
    return nume/float(deno)

def cal_likelihood_laplace(x_train, y_train, feature_col, feature_variation, label):
    x_current = x_train[y_train==label]
    deno = np.sum(y_train==label)
    nume = np.sum(x_current[:,feature_col]==feature_variation)
    m = np.unique(x_train[:,feature_col]).shape[0]
    alpha = 1
    return (nume+alpha)/float(deno+m*alpha)

PREDICT FOR SINGLE DATA¶

# predict a single data without laplace smoothing
def predict(x_train, y_train, x_test):
    classes = np.unique(y_train) 
    num_features = x_train.shape[1]
    prob_set = []
    
    for i in classes:
        prior_prob = cal_prior_prob(i, y_train)

        likelihood = 1.0
        for fe in range(num_features):
            condition_prob = cal_likelihood(x_train, y_train, fe, x_test[fe], i)
            likelihood *= condition_prob
            
        prob = prior_prob * likelihood
        prob_set.append(prob)
    
    result = np.argmax(prob_set)
    return result

# predict for whole testing set
def confusion_matrix(x_train, y_train, x_test, y_test):
    actual_values = y_test.tolist()
    pre_results = []
    
    for i in range(x_test.shape[0]):
        result = predict(x_train, y_train, x_test[i])
        pre_results.append(result)
    
    fp = 0
    fn = 0
    tp = 0
    tn = 0
    
    for actual_value, predicted_result in zip(actual_values, pre_results):
    # let's first see if it's a true (t) or false prediction (f)
        if predicted_result == actual_value: # t?
            if predicted_result == 1: # tp
                tp += 1
            else: # tn
                tn += 1
        else: # f?
            if predicted_result == 1: # fp
                fp += 1
            else: # fn
                fn += 1
    data = {'y_Actual': actual_values,
        'y_Predicted': pre_results
        }

    df_con = pd.DataFrame(data, columns=['y_Actual','y_Predicted'])

    confusion_matrix_nok = pd.crosstab(df_con['y_Actual'], df_con['y_Predicted'], rownames=['Actual'], colnames=['Predicted'])
      
            
    return fp, fn, tp, tn, confusion_matrix_nok
